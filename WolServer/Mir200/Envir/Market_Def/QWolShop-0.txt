;此脚为商城脚本，用于实现各种与商城有关的功能

;商城――白银钥匙
[@BuyShopItem665]
#IF
random 1
#ACT
mov P0 $PARAM(6)
MULT P0 $PARAM(1)

#IF
CHECKGAMEGOLD \ $STR(P0)
#ACT
GAMEGOLD - $STR(P0)
mov P0 2
MULT P0 $PARAM(6)
give 白银钥匙 $STR(P0)
CollectPoint + $STR(P0)
MessageBox ""Congratulations on your purchase success!""
break
